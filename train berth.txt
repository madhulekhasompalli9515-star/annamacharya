import java.util.Scanner;
public class Main
{
	public static void main(String[] args) {
	    Scanner input=new Scanner(System.in);
	    System.out.print("enter berth number:");
	    int berth=input.nextInt();
	    int rem=berth%8;
		switch(rem)
		{
		    case 1:
		    case 4:
		        System.out.println("Lower berth");
		        break;
		    case 2:
		    case 5:
		        System.out.println("Middle berth");
		        break;
		    case 3:
		    case 6:
		        System.out.println("Upper berth");
		        break;
		    case 7:
		        System.out.println("Side lower");
		        break;
		    case 0:
		        System.out.println("Side upper");
		        break;
		    default:
		        System.out.println("enter valid berth");
		    
		}
		    
	}
}
